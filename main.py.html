158  def calculate_equilibrium_constant(reactants, products):
159      reactant_concentration = 1
160      product_concentration = 1
161      for r in reactants:
162          reactant_concentration *= r
163      for p in products:
164          product_concentration *= p
165      equilibrium_constant = product_concentration / reactant_concentration
166      return equilibrium_constant
